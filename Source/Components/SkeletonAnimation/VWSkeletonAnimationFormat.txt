自定义的骨骼动画格式，仅保存需要的最少信息
模型顶点、索引均不在此格式保存的范围内

[Header]

BoneCount = <Integer>
AnimationCount = <Integer>

[Bones]

Name<Index> = <String>
Parent<Index> = <Index | -1>

[Animation<Index>]

Name = <String>

KeyframeCount = <Integer>

Time<Index> = <Float>

Pos<Index> = <Float> <Float> <Float>
Rot<Index> = <Float> <Float> <Float> <Float>
Scl<Index> = <Float> <Float> <Float>
